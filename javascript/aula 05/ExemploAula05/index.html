/* FOR */ // Exemplo de for loop que imprime os números de 1 a 5 console.log('Exemplo: Exemplo de for loop que imprime os números de 1 a 5'); for (let i = 1; i <= 5; i++) { console.log(i); } // Exemplo de for loop que itera sobre os elementos de um array console.log('Exemplo: Exemplo de for loop que itera sobre os elementos de um array'); const colors = ['red', 'green', 'blue']; for (let i = 0; i < colors.length; i++) { console.log(colors[i]); } /* WHILE */ // Exemplo de while loop que imprime os números de 1 a 5 console.log('Exemplo: Exemplo de while loop que imprime os números de 1 a 5'); let counter = 1; while (counter <= 5) { console.log(counter); counter++; } // Exemplo de while loop que imprime números pares menores que 10 console.log('Exemplo: Exemplo de while loop que imprime números pares menores que 10'); let num = 2; while (num < 10) { console.log(num); num += 2; }
